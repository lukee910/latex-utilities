% Title shorthands
\let\h\section
\let\hh\subsection
\let\hhh\subsubsection

% Braces
\newcommand{\braced}[1] {\left( #1 \right)}
\newcommand{\curly}[1] {\left\lbrace #1 \right\rbrace}
\newcommand{\brackets}[1] {\left[ #1 \right]}

% Counters
% Set the counter so that the next header will have the given value
% For use before theorems, titles, etc.
\newcommand{\setnext}[2]{
  \setcounter{#1}{#2}
  \addtocounter{#1}{-1}
}
